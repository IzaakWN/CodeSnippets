% HIGGS PAIR production
\begin{figure}[p]
	\centering
	\begin{fmffile}{ggH1}
	\begin{fmfgraph*}(\wwg,\hh)
    	\fmfset{arrow_len}{\arrowlen}
    	\fmfset{curly_len}{\curlylen}
    	\fmfset{dash_len}{\dashlen}
		\fmfstraight
		\fmfleft{i1,i2}
		\fmfright{o1,m,o2}
		% gluons
		\fmf{gluon}{i1,t1}
		\fmf{gluon}{t2,i2}
		\fmf{phantom,tension=0.4}{t1,o1}
		\fmf{phantom,tension=0.4}{t2,o2}
		\fmffreeze
		% top loop
		\fmf{fermion,tension=1}{t1,t2,t3,t1}
		\fmf{phantom,tension=1.4}{t3,m}
		\fmffreeze
		% Higgs boson
		\fmf{dashes,tension=1.4}{t3,h}
		\fmf{dashes,tension=1}{h,o1}
		\fmf{dashes,tension=1}{h,o2}
%		\fmflabel{\huge\quad$\sim \lambda$}{h}
	\end{fmfgraph*}
	\end{fmffile}
	\vspace{5mm}
	\caption{Feynman diagram for Higgs pair production through gluon-gluon production and trilinear self-coupling.}
	\vspace{5mm}
\end{figure}
