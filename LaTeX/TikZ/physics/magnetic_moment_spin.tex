\documentclass[border=3pt,tikz]{standalone}
\usepackage{physics}
\usetikzlibrary{decorations.markings} % for arrow along path
\tikzset{>=latex} % for LaTeX arrow head
\usetikzlibrary{arrows.meta} % for arrow size

\colorlet{Scol}{green!60!black}
\colorlet{mucol}{red!90!black}
\tikzstyle{mu vector}=[->,thick,mucol,line cap=round]
\tikzstyle{spin}=[->,very thick,Scol]
\tikzstyle{->-}=[decoration={markings,mark=at position 0.53 with {\arrow[scale=0.8,rotate=-2]{>}}},
                 postaction={decorate},rotate=\ang-90,line cap=round]
\tikzstyle{charge+}=[very thin,draw=black,circle,inner sep=0.5,
                     top color=red!50,bottom color=red!90!black!90,shading angle=20]
\tikzstyle{charge-}=[charge+,top color=blue!50,bottom color=blue!80]


\begin{document}


% MAGNETIC MOMENT ELECTRON + SPIN
\def\re{0.30} % electron radius
\def\rx{0.45} % horizontal radius of rotation arrow
\def\ry{0.18} % vertical radius of rotation arrow
\def\ang{80}  % tilt
\begin{tikzpicture}
  \draw[rotate=\ang-90] (0,0) ellipse({\rx} and \ry);
  \draw[mu vector] (0,0) -- (\ang+180:3*\re) node[right] {$\vb*{\mu}$}; %_\mathrm{e}
  \draw[spin] (0,0) -- (\ang:2.8*\re) node[right] {$\vb{S}$};
  \draw[charge-]
    (0,0) circle (\re) node[scale=1.2] {$-$};
  \draw[->-]
    (-175:{\rx} and \ry) arc (-175:-35:{\rx} and \ry);
\end{tikzpicture}


% MAGNETIC MOMENT POSITRON + SPIN
\begin{tikzpicture}
  \draw[rotate=\ang-90] (0,0) ellipse({\rx} and \ry);
  \draw[mu vector] (-0.28*\re,0) --++ (\ang:3*\re) node[below=3,left=0] {$\vb*{\mu}$}; %_\mathrm{e}
  \draw[spin] (0.28*\re,0) --++ (\ang:2.8*\re) node[right] {$\vb{S}$};
  \draw[charge+]
    (0,0) circle (\re) node[scale=1.2] {$+$};
  \draw[->-]
    (-175:{\rx} and \ry) arc (-175:-35:{\rx} and \ry);
\end{tikzpicture}


% MAGNETIC MOMENT ELECTRON
\begin{tikzpicture}

    (0,0) circle (\re) node[scale=1.2] {$-$};
