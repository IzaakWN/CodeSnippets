\def\dpa{0.28}
\def\dba{0.14}
\def\dipole#1#2{
  \begin{scope}[shift={(#1)},rotate=#2]
    \draw[charge-] (-\dpa,0) to[out=90,in=180] (0,\dba) -- (0,-\dba) to[out=180,in=-90] cycle;
    \draw[charge+] ( \dpa,0) to[out=90,in=  0] (0,\dba) -- (0,-\dba) to[out=  0,in=-90] cycle;
    \node[scale=0.7] at (-\dpa/2,0) {$-$};
    \node[scale=0.7] at ( \dpa/2,0) {$+$};
  \end{scope}
}
