      \def\RB{0.12}

    code={
      \def\RB{0.12}

  pics/Bin/.default=Bcol,
  pics/spin/.style={
    code={
      \def\L{0.38}

      \draw[-{Latex[length=1.2,width=1]},pic actions,rotate=#1,very thin]
        (0.35*\L,0)++(25:{0.16*\L} and {0.22*\L}) arc (25:305:{0.16*\L} and {0.22*\L}) --++ (50:0.09*\L);
  }},
  pics/spin/.default=90,
