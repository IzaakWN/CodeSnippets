\def\xmax{5.5}
\def\ymax{4}
\def\tick#1#2{\draw[thick] (#1) ++ (#2:0.03*\ymax) --++ (#2-180:0.06*\ymax)}


\begin{document}


% ELECTRIC FIELD of a CHARGED SPHERE
\begin{tikzpicture}
  
  \def\kQ{6.3}
  \def\R{2.0}
  \coordinate (O) at (0,0);
  \coordinate (X) at (\xmax,0);
  \coordinate (Y) at (0,\ymax);
  \coordinate (P) at (\R,\kQ/\R);
  \coordinate (Px) at (\R,0);
  \coordinate (Py) at (0,\kQ/\R);
  
  % PLOT
  \draw[VField,samples=100,smooth,variable=\x,domain=\R:0.98*\xmax]
    (Py) -- plot(\x,\kQ/\x);
  \draw[VField]
    (Py) --++ (Px);
  \node[above right] at (3.5,1.6) {$E \sim \dfrac{1}{r}$};
  \draw[dashed]
    (Px) -- (P);
  
  % AXIS
  \draw[<->,thick]
    (X) node[below] {$r$} -- (O) -- (Y) node[left] {$V$};
  \tick{Py}{ 0} node[below=-1,left] {$\dfrac{kQ}{R}$};
  \tick{Px}{90} node[below] {$R$};
  
\end{tikzpicture}
