  ({(#1-0.22)*\xmax},0) to[out=0,in=180,looseness=0.6] (#1*\xmax,#2*\ymax)

}

  %\draw[wood] (0,-1.3*\ymax) rectangle (\t,1.3*\ymax);

  \draw[wood] (#1+\t/2,1.3*\ymax) ellipse ({\t/2} and {0.4*\t/2});
}
